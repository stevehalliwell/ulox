fun InnerMain()
{
    a *= 2;
}

var a = 0;

test VmLibraryInherTests
{
    testcase InherAndCopyBack
    {
        var expected = 20;
        var startingValue = 10;
        var result = 0;
        var innerVM = VM();

        a = startingValue;
        innerVM.InheritFromEnclosing();
        innerVM.Start(InnerMain);
        innerVM.CopyBackToEnclosing();
        result = a;

        Assert.AreEqual(expected, result);
    }
}
