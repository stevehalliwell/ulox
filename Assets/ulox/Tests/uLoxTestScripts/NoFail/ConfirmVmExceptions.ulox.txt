test ConfirmVmExceptions
{
    testcase CanOnlySetOnClassAndInstance
    {
        fun InvalidSet()
        {
            var a = 3;
            a.property = 7;
        }

        Assert.Throws(InvalidSet);
    }

    testcase CanOnlyGetOnClassAndInstance
    {
        fun InvalidGet()
        {
            var a = 3;
            var b = a.property;
        }

        Assert.Throws(InvalidGet);
    }

    testcase CanOnlyInvokeExtentItems
    {
        fun InvalidInvoke()
        {
            var a = Dynamic();
            a.Invoke();
        }

        Assert.Throws(InvalidInvoke);
    }

    testcase CanOnlyBindOnInstanceOrClass
    {
        fun InvalidInvoke()
        {
            var a = 3;
            var b = a.Meth;
        }

        Assert.Throws(InvalidInvoke);
    }

    testcase InvalidInitArgCount
    {
        fun InvalidInit()
        {
            var a = Dynamic(3);
        }

        Assert.Throws(InvalidInit);
    }

    testcase InvalidMathOverload
    {
        fun InvalidMathOp()
        {
            var a = Dynamic();
            var b = Dynamic();
            a += b;
        }

        Assert.Throws(InvalidMathOp);
    }

    testcase InvalidCompareOverload
    {
        fun InvalidCompare()
        {
            var a = Dynamic();
            var b = Dynamic();
            var res = a < b;
        }

        Assert.Throws(InvalidCompare);
    }
}
